SELECT I.DESCRIPTION, IL.NAME AS LISTNAME
FROM ISSUES I
JOIN ISSUESLISTS IL ON I.ISSUESLIST_ID=IL.ID;

SELECT U.FIRSTNAME, U.LASTNAME, I.DESCRIPTION AS ISSUE
FROM USERS U
JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_QUANTITY
FROM USERS U JOIN ISSUES I ON I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY U.ID
HAVING COUNT(*) >0;
